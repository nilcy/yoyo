6   package yoyo.framework.enterprise.infra.persistence;
7   import java.io.Serializable;
8   import java.util.Collection;
9   import java.util.Map;
10  import javax.persistence.LockModeType;
11  import javax.persistence.TypedQuery;
12  import yoyo.framework.enterprise.shared.Persistable;
13  import yoyo.framework.standard.shared.Service;
14  /**
15   * 基本データ永続化サービスI/F
16   * <ul>
17   * エンティティのライフサイクル
18   * <li>New(新規)…インスタンス化しただけの状態。[persist()でManaged化]</li>
19   * <li>Managed(管理)…永続コンテキスト中で管理している状態。[find()が可能]</li>
20   * <li>Removed(削除)…永続コンテキスト中で削除を予約している状態。[remove()した後]</li>
21   * <li>Detached(分離)…永続コンテキストから分離した状態。[merge()でManaged化]</li>
22   * </ul>
23   * @param <E> エンティティ型
24   * @param <ID> 識別子オブジェクト型
25   * @author nilcy
26   */
27  public interface SimplePersistenceService<E extends Persistable<ID>, ID extends Serializable>
28      extends Service {
29      /**
30       * 管理エンティティの永続化
31       * <p>
32       * エンティティ状態の遷移は「新規→管理、削除→管理、管理→(無視)」となる。分離のとき EntityExistsException 例外を発生する。
33       * </p>
34       * @param entity エンティティ
35       */
36      <S extends E> void persist(S entity);
37      /**
38       * 管理エンティティのID検索
39       * @param id 識別子
40       * @return 管理エンティティ
41       */
42      E find(ID id);
43      /**
44       * 管理エンティティのID検索
45       * @param id 識別子
46       * @param lockModeType ロックモードタイプ
47       * @return 管理エンティティ
48       */
49      E find(ID id, LockModeType lockModeType);
50      /**
51       * 管理エンティティのID検索
52       * @param id 識別子
53       * @param properties プロパティ
54       * @return 管理エンティティ
55       */
56      E find(final ID id, final Map<String, Object> properties);
57      /**
58       * 管理エンティティのID検索
59       * @param id 識別子
60       * @param lockModeType ロックモードタイプ
61       * @param properties プロパティ
62       * @return 管理エンティティ
63       */
64      E find(ID id, LockModeType lockModeType, final Map<String, Object> properties);
65      /**
66       * 分離エンティティの永続化
67       * <p>
68       * エンティティ状態の遷移は「新規→管理、管理→(無視)、分離→管理」となる。削除のとき IllegalArgumentException 例外を発生する。
69       * </p>
70       * @param entity 分離エンティティ
71       * @return 管理エンティティ
72       */
73      <S extends E> S merge(S entity);
74      /**
75       * 管理エンティティの削除
76       * <p>
77       * エンティティ状態の遷移は「新規→(無視)、管理→削除、削除→(無視)」となる。分離のとき IllegalArgumentException 例外を発生する。
78       * </p>
79       * @param entity 管理エンティティ
80       */
81      void remove(E entity);
82      /**
83       * 管理エンティティの更新
84       * @param entity 管理エンティティ
85       */
86      void refresh(final E entity);
87      /**
88       * 管理エンティティの更新
89       * @param entity 管理エンティティ
90       * @param lockModeType 保護モード
91       */
92      void refresh(final E entity, final LockModeType lockModeType);
93      /**
94       * 管理エンティティの更新
95       * @param entity 管理エンティティ
96       * @param properties プロパティ
97       */
98      void refresh(final E entity, final Map<String, Object> properties);
99      /**
100      * 管理エンティティの更新
101      * @param entity 管理エンティティ
102      * @param lockModeType 保護モード
103      * @param properties プロパティ
104      */
105     void refresh(final E entity, final LockModeType lockModeType,
106         final Map<String, Object> properties);
107     /**
108      * 管理エンティティの保護
109      * @param entity 管理エンティティ
110      * @param lockModeType 保護モード
111      */
112     void lock(final E entity, LockModeType lockModeType);
113     /**
114      * 管理エンティティの保護
115      * @param entity 管理エンティティ
116      * @param lockModeType 保護モード
117      * @param properties プロパティ
118      */
119     void lock(final E entity, LockModeType lockModeType, final Map<String, Object> properties);
120     /** 管理エンティティのDB反映 */
121     void flush();
122     /**
123      * 管理エンティティの分離
124      * @param entity 管理エンティティ
125      */
126     void detach(E entity);
127     /**
128      * 管理エンティティ含有の確認
129      * @param entity エンティティ
130      * @return 含有するとき真。含有しないとき偽。
131      */
132     boolean contains(final E entity);
133     /**
134      * 範囲指定クエリの作成
135      * @param query クエリ
136      * @param offset 開始位置
137      * @param maxsize 最大件数
138      * @return クエリ
139      */
140     TypedQuery<E> createRangeQuery(final TypedQuery<E> query, final int offset, final int maxsize);
141     /**
142      * 管理エンティティ集合のクエリ検索
143      * @param query クエリ
144      * @return 管理エンティティ集合
145      */
146     Collection<E> findMany(TypedQuery<E> query);
147     /**
148      * 管理エンティティのクエリ検索
149      * @param query クエリ
150      * @return 管理エンティティ
151      */
152     E findOne(TypedQuery<E> query);
153     /**
154      * プロパティの取得
155      * @return プロパティ
156      */
157     Map<String, Object> getProperties();
158     /**
159      * プロパティの設定
160      * @param propertyName プロパティ名
161      * @param value プロパティ値
162      */
163     void setProperty(String propertyName, Object value);
164 }
